198 // Powers off the IoT Node board using the RTC
199 // RTC CONTROL switch must be set to Yes
200 void IoTNode::switchOffFor(long seconds, maskValue mask)
201 {
202  // Set the RTC high so that the power stays on until the alarm is enabled
203  rtc.outHigh();
204  // Disable both alarms
205  rtc.disableAlarms();
206  // Set the timer mask
207  rtc.maskAlarm0(mask);
208  // Set the alarm polarization high - i.e. when alarm sets switch back on
209  rtc.setAlarm0PolHigh();
210  rtc.clearIntAlarm0();
211  // Load the alarm match value (all registers)
212  int rtcnow = rtc.rtcNow();
213  int alarmTime = rtcnow + seconds;
214  rtc.setAlarm0UnixTime(alarmTime);
215  // Enable the alarm. This will set the MFP output low
216  // turning off the power until the alarm triggers
217  // turning on the power again and starting the code from
218  // the beginning.
219  rtc.enableClock();
220  // Ensure the ALMxIF flag is cleared
221  rtc.enableAlarm0();
222  Wire.end();
223  delay(200);
224 }

388 void IoTNode::array_to_string(byte array[], unsigned int len, char buffer[])
389 {
390  for (unsigned int i = 0; i < len; i++)
391  {
392  byte nib1 = (array[i] >> 4) & 0x0F;
393  byte nib2 = (array[i] >> 0) & 0x0F;
394  buffer[i*2+0] = nib1 < 0xA ? '0' + nib1 : 'A' + nib1 - 0xA;
395  buffer[i*2+1] = nib2 < 0xA ? '0' + nib2 : 'A' + nib2 - 0xA;
396  }
397  buffer[len*2] = '\0';
398 }
